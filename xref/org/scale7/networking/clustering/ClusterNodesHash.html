51  	 * Set the nodes that comprise the cluster. This is performed as a single operation. 
52  	 * @param clusterNodes The nodes that comprise the cluster
53  	 * @return A set containing the nodes objects that were previously in the cluster, but which have been removed
54  	 */
55  	public Set<T> setMembers(Collection<T> clusterNodes) {
56  		SortedMap<Long, T> newRing = new TreeMap<Long, T>();
57  		HashMap<String, T> newNodeMap = new HashMap<String, T>();
58  
59  		for (T node : clusterNodes) {
60  			addNode(newRing, newNodeMap, node, replicationFactor);
61  		}	
62  
63  		HashSet<T> removedNodes;
64  		if (ring == null) {
65  			removedNodes = new HashSet<T>();
66  		} else {
67  			removedNodes = new HashSet<T>(ring.values());
68  			removedNodes.removeAll(newRing.values());
69  		}
70  		
71  		ring = newRing;
72  		nodeMap = newNodeMap;
73  		
74  		return removedNodes;
75  	}

92  	public T getNodeByResponsibility(String objectId) {
93  		SortedMap<Long, T> currRing = ring;

95  		if (currRing.isEmpty()) {

97  		}
98  		long hash = hasher.hash(objectId);
99  		if (!currRing.containsKey(hash)) {
100 			SortedMap<Long, T> tailMap = currRing.tailMap(hash);
101 			hash = tailMap.isEmpty() ? currRing.firstKey() : tailMap.firstKey();
102 		}
103 		return currRing.get(hash);
104 	}

106 	private void addNode(SortedMap<Long, T> circle, AbstractMap<String, T> nodeMap, T node, int replicationFactor) {
107 		for (int i = 0; i < replicationFactor; i++) {
108 			circle.put(hasher.hash(node.toString() + i), node);
109 		}

124 		public HashCalculator(String hashName) throws NoSuchAlgorithmException {
125 			md = MessageDigest.getInstance(hashName);
126 		}
127 		
128 		long hash(String key) {
129 			md.reset();
130 			md.update(key.getBytes());
131 			byte[] digest = md.digest();
132 			
133 	        long h = 0;
134 	        for (int i = 0; i < 4; i++) {
135 	            h <<= 8;
136 	            h |= ((int) digest[i]) & 0xFF;
137 	        }
138 	        return h;			
139 		}
